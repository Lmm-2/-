<script>
    let isAllExpanded = true;

    function enterSite() {
        document.getElementById('welcome-screen').classList.add('hidden');
    }

    function toggleTheme() {
        document.body.classList.toggle('dark-mode');
        const themeBtn = document.getElementById('themeBtn');
        themeBtn.textContent = document.body.classList.contains('dark-mode') ? '☀️' : '🌙';
    }

    function toggleBranch(headerElem) {
        const card = headerElem.closest('.branch-card');
        card.classList.toggle('collapsed');
    }

    function toggleAllBranches() {
        const cards = document.querySelectorAll('.branch-card');
        if (isAllExpanded) {
            cards.forEach(card => card.classList.add('collapsed'));
            isAllExpanded = false;
        } else {
            cards.forEach(card => card.classList.remove('collapsed'));
            isAllExpanded = true;
        }
    }

    function highlightBranch(id) {
        const element = document.getElementById(id);
        if (element) {
            element.classList.remove('collapsed');
            element.classList.add('highlight');
            setTimeout(() => {
                element.classList.remove('highlight');
            }, 1500);
        }
    }

    function calculateStats() {
        const cards = document.querySelectorAll('.branch-card');
        document.getElementById('branches-count').textContent = cards.length;

        const memberNames = document.querySelectorAll('.member-name, .sub-item, .branch-title-group h2');
        document.getElementById('members-count').textContent = memberNames.length + 2;
    }

    window.onload = calculateStats;
</script>
